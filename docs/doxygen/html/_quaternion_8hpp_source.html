117  void dcmToQuat(const NdArray<double>& dcm)
118  {
119  const Shape inShape = dcm.shape();
120  if (!(inShape.rows == 3 && inShape.cols == 3))
121  {
122  THROW_INVALID_ARGUMENT_ERROR("input direction cosine matrix must have shape = (3,3).");
123  }
124 
125  NdArray<double> checks(1, 4);
126  checks[0] = 1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2);
127  checks[1] = 1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2);
128  checks[2] = 1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2);
129  checks[3] = 1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2);
130 
131  const uint32 maxIdx = argmax(checks).item();
132 
133  switch (maxIdx)
134  {
135  case 0:
136  {
137  components_[3] = 0.5 * std::sqrt(1 + dcm(0, 0) + dcm(1, 1) + dcm(2, 2));
138  components_[0] = (dcm(2, 1) - dcm(1, 2)) / (4 * components_[3]);
139  components_[1] = (dcm(0, 2) - dcm(2, 0)) / (4 * components_[3]);
140  components_[2] = (dcm(1, 0) - dcm(0, 1)) / (4 * components_[3]);
141 
142  break;
143  }
144  case 1:
145  {
146  components_[0] = 0.5 * std::sqrt(1 + dcm(0, 0) - dcm(1, 1) - dcm(2, 2));
147  components_[1] = (dcm(1, 0) + dcm(0, 1)) / (4 * components_[0]);
148  components_[2] = (dcm(2, 0) + dcm(0, 2)) / (4 * components_[0]);
149  components_[3] = (dcm(2, 1) - dcm(1, 2)) / (4 * components_[0]);
150 
151  break;
152  }
153  case 2:
154  {
155  components_[1] = 0.5 * std::sqrt(1 - dcm(0, 0) + dcm(1, 1) - dcm(2, 2));
156  components_[0] = (dcm(1, 0) + dcm(0, 1)) / (4 * components_[1]);
157  components_[2] = (dcm(2, 1) + dcm(1, 2)) / (4 * components_[1]);
158  components_[3] = (dcm(0, 2) - dcm(2, 0)) / (4 * components_[1]);
159 
160  break;
161  }
162  case 3:
163  {
164  components_[2] = 0.5 * std::sqrt(1 - dcm(0, 0) - dcm(1, 1) + dcm(2, 2));
165  components_[0] = (dcm(2, 0) + dcm(0, 2)) / (4 * components_[2]);
166  components_[1] = (dcm(2, 1) + dcm(1, 2)) / (4 * components_[2]);
167  components_[3] = (dcm(1, 0) - dcm(0, 1)) / (4 * components_[2]);
168 
169  break;
170  }
171  }

285  static NdArray<double> angularVelocity(const Quaternion& inQuat1, const Quaternion& inQuat2, double inTime) noexcept
286  {
287  NdArray<double> q0 = inQuat1.toNdArray();
288  NdArray<double> q1 = inQuat2.toNdArray();
289 
290  NdArray<double> qDot = q1 - q0;
291  qDot /= inTime;
292 
293  NdArray<double> eyeTimesScalar(3);
294  eyeTimesScalar.zeros();
295  eyeTimesScalar(0, 0) = inQuat2.s();
296  eyeTimesScalar(1, 1) = inQuat2.s();
297  eyeTimesScalar(2, 2) = inQuat2.s();
298 
299  NdArray<double> epsilonHat = linalg::hat<double>(inQuat2.i(), inQuat2.j(), inQuat2.k());
300  NdArray<double> q(4, 3);
301  q.put(Slice(0, 3), Slice(0, 3), eyeTimesScalar + epsilonHat);
302  q(3, 0) = -inQuat2.i();
303  q(3, 1) = -inQuat2.j();
304  q(3, 2) = -inQuat2.k();
305 
306  NdArray<double> omega = q.transpose().dot(qDot.transpose());
307  return omega *= 2.0;

538  static Quaternion slerp(const Quaternion& inQuat1, const Quaternion& inQuat2, double inPercent)
539  {
540  if (inPercent < 0 || inPercent > 1)
541  {
542  THROW_INVALID_ARGUMENT_ERROR("input percent must be of the range [0, 1]");
543  }
544 
545  if (inPercent == 0)
546  {
547  return inQuat1;
548  }
549  else if (inPercent == 1)
550  {
551  return inQuat2;
552  }
553 
554  double dotProduct = dot<double>(inQuat1.toNdArray(), inQuat2.toNdArray()).item();
555 
556  // If the dot product is negative, the quaternions
557  // have opposite handed-ness and slerp won't take
558  // the shorter path. Fix by reversing one quaternion.
559  Quaternion quat1Copy(inQuat1);
560  if (dotProduct < 0.0)
561  {
562  quat1Copy *= -1;
563  dotProduct *= -1;
564  }
565 
566  const double DOT_THRESHOLD = 0.9995;
567  if (dotProduct > DOT_THRESHOLD) {
568  // If the inputs are too close for comfort, linearly interpolate
569  // and normalize the result.
570  return nlerp(inQuat1, inQuat2, inPercent);
571  }
572 
573  dotProduct = clip(dotProduct, -1.0, 1.0); // Robustness: Stay within domain of acos()
574  const double theta0 = std::acos(dotProduct); // angle between input vectors
575  const double theta = theta0 * inPercent; // angle between v0 and result
576 
577  const double s0 = std::cos(theta) - dotProduct * std::sin(theta) / std::sin(theta0); // == sin(theta_0 - theta) / sin(theta_0)
578  const double s1 = std::sin(theta) / std::sin(theta0);
579 
580  NdArray<double> interpQuat = (inQuat1.toNdArray() * s0) + (inQuat2.toNdArray() * s1);
581  return Quaternion(interpQuat);
582  }

626  const double q2 = k();
627  const double q3 = s();

638  dcm(1, 1) = q3sqr + q1sqr - q0sqr - q2sqr;
639  dcm(1, 2) = 2 * (q1 * q2 - q3 * q0);
640  dcm(2, 0) = 2 * (q0 * q2 - q3 * q1);
641  dcm(2, 1) = 2 * (q1 * q2 + q3 * q0);
642  dcm(2, 2) = q3sqr + q2sqr - q0sqr - q1sqr;
